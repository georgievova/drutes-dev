592  subroutine get2dderivative(a,b,c,xder,yder)
593  use typy
595  real(kind=rkind), dimension(:), intent(in) :: a
597  real(kind=rkind), dimension(:), intent(in) :: b
599  real(kind=rkind), dimension(:), intent(in) :: c

603  real(kind=rkind), intent(out) :: yder
604  !-------local variables--------------
605  real(kind=rkind), dimension(3) :: u
606  real(kind=rkind), dimension(3) :: v
607  real(kind=rkind), dimension(3) :: n
608  integer(kind=ikind) :: i
609  real(kind=rkind) :: reps
610 
611 
612  reps = epsilon(reps)
613 
614 
615  !check if the plane is not horizontal
616  if (abs(a(3) - b(3)) < reps*(abs(a(3))-abs(b(3))) .and. &
617  abs(a(3) - c(3)) < reps*(abs(a(3))-abs(c(3))) .and. &
618  abs(b(3) - c(3)) < reps*(abs(b(3))-abs(c(3)))) then
619  xder = 0.0_rkind
620  yder = 0.0_rkind
621  RETURN
622  else
623  CONTINUE
624  end if
625 
626  !creates the plane vectors
627  do i=1,3
628  u(i) = a(i) - b(i)
629  v(i) = a(i) - c(i)
630  end do
631 
632  ! the normal plane vector is defined as
633  n(1) = u(2)*v(3) - v(2)*u(3)
634  n(2) = u(3)*v(1) - v(3)*u(1)
635  n(3) = u(1)*v(2) - v(1)*u(2)
636 
637  ! finally the derivate is as follows, the horizontality has been already checked
638  ! the verticality check
639  if (abs(n(3)) < 1e2*reps) then
640  print *, "the mesh is wrong, base function can't be vertical"
641  error stop
642  end if
643  xder = -n(1)/n(3)
644  yder = -n(2)/n(3)
